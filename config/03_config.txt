server {
  port: 8080;
  server_names: localhost, site1.com, www.site1.com;
  root: www/;
  default_pages: index.html, home.html;
  error_pages {
    404: default/errors/404.html;
  }
  auto_index: true;
  allowed_methods: GET, POST, DELETE;
  max_body_bytes: 1M;
  cgi_path {
    .php: /usr/bin/php-cgi;
    .py: /usr/bin/python3;
  }
  redirect: welcome/;
  upload: uploads/;
  location: /admin {
    root: /var/www/site1/admin/;
    allowed_methods: GET, POST;
  }
}
